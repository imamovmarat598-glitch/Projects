// Upload process
async function uploadPhoto(file: File, manualGPS?: { lat: number, lon: number }) {
  // 1. Validate file
  if (file.size > 10 * 1024 * 1024) {
    throw new Error('File too large (max 10MB)');
  }

  // 2. Extract EXIF GPS
  const exif = await exifr.parse(file);
  const gps = exif?.latitude && exif?.longitude
    ? { lat: exif.latitude, lon: exif.longitude }
    : manualGPS;

  if (!gps) {
    throw new Error('No GPS coordinates (add manually)');
  }

  // 3. Generate thumbnail (client-side)
  const thumbnail = await generateThumbnail(file, { maxWidth: 640, maxHeight: 480 });

  // 4. Prepare metadata
  const metadata = {
    device_id: generateWebDeviceId(),
    timestamp: new Date().toISOString(),
    gps,
    filename: file.name,
    size: file.size,
    format: file.type,
  };

  // 5. Upload to backend
  const formData = new FormData();
  formData.append('photo', file);
  formData.append('thumbnail', thumbnail);
  formData.append('metadata', JSON.stringify(metadata));

  const response = await fetch('/api/photos/upload', {
    method: 'POST',
    body: formData,
    // Progress tracking
    onUploadProgress: (progress) => {
      updateProgressBar(progress.loaded / progress.total);
    },
  });

  return response.json();
}
